            const findGreatestNumber = (num1, num2, num3) => {
              let arr = [num1, num2, num3];
              let result = 0;
              arr.forEach((num) => {
                num > result && (result = num);
              });
              return result;
            };

            console.log('Greatest number is:', findGreatestNumber(num1, num2, num3));
